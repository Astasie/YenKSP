00027 class priorityDictionary(dict):
00028     def __init__(self):
00029         '''Initialize priorityDictionary by creating binary heap of pairs 
00030         (value,key).  Note that changing or removing a dict entry will not 
00031         remove the old pair from the heap until it is found by smallest() or
00032                 until the heap is rebuilt.'''
00033         self.__heap = []
00034         dict.__init__(self)
00035 
00036     def smallest(self):
00037         '''Find smallest item after removing deleted items from heap.'''
00038         if len(self) == 0:
00039             raise IndexError, "smallest of empty priorityDictionary"
00040         heap = self.__heap
00041         while heap[0][1] not in self or self[heap[0][1]] != heap[0][0]:
00042             lastItem = heap.pop()
00043             insertionPoint = 0
00044             while 1:
00045                 smallChild = 2*insertionPoint+1
00046                 if smallChild+1 < len(heap) and \
00047                         heap[smallChild] > heap[smallChild+1]:
00048                     smallChild += 1
00049                 if smallChild >= len(heap) or lastItem <= heap[smallChild]:
00050                     heap[insertionPoint] = lastItem
00051                     break
00052                 heap[insertionPoint] = heap[smallChild]
00053                 insertionPoint = smallChild
00054         return heap[0][1]
00055         
00056     def __iter__(self):
00057         '''Create destructive sorted iterator of priorityDictionary.'''
00058         def iterfn():
00059             while len(self) > 0:
00060                 x = self.smallest()
00061                 yield x
00062                 del self[x]
00063         return iterfn()
00064         
00065     def __setitem__(self,key,val):
00066         '''Change value stored in dictionary and add corresponding pair to heap.  
00067         Rebuilds the heap if the number of deleted items grows too large, to 
00068         avoid memory leakage.'''
00069         dict.__setitem__(self,key,val)
00070         heap = self.__heap
00071         if len(heap) > 2 * len(self):
00072             self.__heap = [(v,k) for k,v in self.iteritems()]
00073             self.__heap.sort()  # builtin sort likely faster than O(n) heapify
00074         else:
00075             newPair = (val,key)
00076             insertionPoint = len(heap)
00077             heap.append(None)
00078             while insertionPoint > 0 and \
00079                     newPair < heap[(insertionPoint-1)//2]:
00080                 heap[insertionPoint] = heap[(insertionPoint-1)//2]
00081                 insertionPoint = (insertionPoint-1)//2
00082             heap[insertionPoint] = newPair
00083         
00084     def setdefault(self,key,val):
00085         '''Reimplement setdefault to call our customized __setitem__.'''
00086         if key not in self:
00087             self[key] = val
00088         return self[key]
