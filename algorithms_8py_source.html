00029 ## @package YenKSP
00030 # Computes K-Shortest Paths using Yen's Algorithm.
00031 #
00032 # Yen's algorithm computes single-source K-shortest loopless paths for a graph 
00033 # with non-negative edge cost. The algorithm was published by Jin Y. Yen in 1971
00034 # and implores any shortest path algorithm to find the best path, then proceeds 
00035 # to find K-1 deviations of the best path.
00036 
00037 ## Computes K paths from a source to a sink in the supplied graph.
00038 #
00039 # @param graph A digraph of class Graph.
00040 # @param start The source node of the graph.
00041 # @param sink The sink node of the graph.
00042 # @param K The amount of paths being computed.
00043 #
00044 # @retval [] Array of paths, where [0] is the shortest, [1] is the next 
00045 # shortest, and so on.
00046 #
00047 def ksp_yen(graph, node_start, node_end, max_k=2):
00048     distances, previous = dijkstra(graph, node_start)
00049     
00050     A = [{'cost': distances[node_end], 
00051           'path': path(previous, node_start, node_end)}]
00052     B = []
00053     
00054     for k in range(1, max_k):
00055         for i in range(0, len(A[-1]['path']) - 1):
00056             node_spur = A[-1]['path'][i]
00057             path_root = A[-1]['path'][:i+1]
00058             
00059             edges_removed = []
00060             for path_k in A:
00061                 if path_root == path_k['path'][:i+1]:
00062                     cost = graph.remove_edge(path_k['path'][i], 
00063                                              path_k['path'][i+1])
00064                     if cost == -1:
00065                         continue
00066                     edges_removed.append([path_k['path'][i], 
00067                                           path_k['path'][i+1], 
00068                                           cost])
00069             
00070             path_spur = dijkstra(graph, node_spur, node_end)
00071             
00072             if path_spur:
00073                 path_total = path_root[:-1] + path_spur['path']
00074                 dist_total = distances[node_spur] + path_spur['cost']
00075                 potential_k = {'cost': dist_total, 'path': path_total}
00076             
00077                 if not (potential_k in B):
00078                     B.append(potential_k)
00079             
00080             for edge in edges_removed:
00081                 graph.add_edge(edge[0], edge[1], edge[2])
00082         
00083         if len(B):
00084             B = sorted(B, key=itemgetter('cost'))
00085             A.append(B[0])
00086             B.pop(0)
00087         else:
00088             break
00089     
00090     return A

00092 ## Computes the shortest path from a source to a sink in the supplied graph.
00093 #
00094 # @param graph A digraph of class Graph.
00095 # @param node_start The source node of the graph.
00096 # @param node_end The sink node of the graph.
00097 #
00098 # @retval {} Dictionary of path and cost or if the node_end is not specified,
00099 # the distances and previous lists are returned.
00100 #
00101 def dijkstra(graph, node_start, node_end=None):
00102     distances = {}      
00103     previous = {}       
00104     Q = priorityDictionary()
00105     
00106     for v in graph:
00107         distances[v] = graph.INFINITY
00108         previous[v] = graph.UNDEFINDED
00109         Q[v] = graph.INFINITY
00110     
00111     distances[node_start] = 0
00112     Q[node_start] = 0
00113     
00114     for v in Q:
00115         if v == node_end: break
00116 
00117         for u in graph[v]:
00118             cost_vu = distances[v] + graph[v][u]
00119             
00120             if cost_vu < distances[u]:
00121                 distances[u] = cost_vu
00122                 Q[u] = cost_vu
00123                 previous[u] = v
00124 
00125     if node_end:
00126         return {'cost': distances[node_end], 
00127                 'path': path(previous, node_start, node_end)}
00128     else:
00129         return (distances, previous)

00131 ## Finds a paths from a source to a sink using a supplied previous node list.
00132 #
00133 # @param previous A list of node predecessors.
00134 # @param node_start The source node of the graph.
00135 # @param node_end The sink node of the graph.
00136 #
00137 # @retval [] Array of nodes if a path is found, an empty list if no path is 
00138 # found from the source tp sink.
00139 #
00140 def path(previous, node_start, node_end):
00141     route = []
00142 
00143     node_curr = node_end    
00144     while True:
00145         route.append(node_curr)
00146         if previous[node_curr] == node_start:
00147             route.append(node_start)
00148             break
00149         elif previous[node_curr] == DiGraph.UNDEFINDED:
00150             return []
00151         
00152         node_curr = previous[node_curr]
00153     
00154     route.reverse()
00155     return route
